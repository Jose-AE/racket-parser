    int grade = 75;

    switch (grade) {
    case 90:
        printf("A");
        break;
    case 80:
        printf("B");
        break;
    default:
        printf("C or below");
    }

    while (grade < 80) {
        printf("Grade");
        grade++;
    }

        printf("Grade");
        grade++;
    } while (grade < 85);

        printf("Grade is greater than or equal to 85");
    }

            a++;

        }
        printf("a");
        a++;
    }

        if (b == 1) {
            b++;
            continue;
        }
        printf("b");
        b++;
    } while (b < 5);

    int x = 0;
